11. Bản đồ với folium

11.1. Giới thiệu

Trong python, ta có thể sử dụng package folium để tương tác và làm việc với leaflet để xây dựng bản đồ động.

# Tạo bản đồ leaflet
import polars as pl
import folium
from folium.plugins import MarkerCluster, HeatMap
import random

# Set center location (Hanoi)
center_lat, center_lon = 21.0285, 105.8542

# Generate 500 random points around Hanoi
random.seed(42)
latitudes = [center_lat + random.uniform(-0.02, 0.02) for _ in range(500)]
longitudes = [center_lon + random.uniform(-0.02, 0.02) for _ in range(500)]

# Create Polars DataFrame
df = pl.DataFrame({
    "lat": latitudes,
    "lon": longitudes
})

# %%

# Create map
m = folium.Map(location=[center_lat, center_lon], zoom_start=13, tiles='CartoDB positron')
# --- Marker Cluster with CircleMarkers ---
marker_cluster = MarkerCluster(name="Cluster View").add_to(m)

for row in df.iter_rows():
    folium.CircleMarker(
        location=[row[0], row[1]],
        radius=6,
        color='blue',
        fill=False,
        fill_color='blue',
        fill_opacity=0.2,
        popup=f"Lat: {row[0]:.4f}, Lon: {row[1]:.4f}"
    ).add_to(marker_cluster)

# --- Heatmap ---
heat_data = df.select(["lat", "lon"]).to_numpy().tolist()
HeatMap(heat_data, name="Heatmap View", radius=10).add_to(m)

# Layer toggle
folium.LayerControl().add_to(m)
